version: "3.8"

services:
  postgres:
    image: postgres:15
    container_name: event-db
    restart: always
    environment:
      POSTGRES_USER: user
      POSTGRES_PASSWORD: password
      POSTGRES_DB: events
    ports:
      - "5432:5432"

  backend:
    build:
      context: ..
      dockerfile: docker/Dockerfile
    container_name: event-backend
    ports:
      - "5000:5000"
    depends_on:
      - postgres
    env_file:
      - ../.env
    volumes:
      - ..:/app
    command: npm run dev
